Scientific Figures with Matplotlib

To run the notebook below you need to have matplotlib, seaborn, numpy and scipy installed. If needed use

pip install matplotlib numpy seaborn scipy

to install.

Note: matplotlib 3.6.2, seaborn 0.12.2 were used when writing this code. Some functions may have changed since then.

DPI vs Figsize

Example taken from stackoverflow

import matplotlib.pyplot as plt
%matplotlib inline

def plot(fs,dpi, savepath):
    fig, ax=plt.subplots(figsize=fs, dpi=dpi)
    ax.set_title("Figsize: {}, dpi: {}".format(fs,dpi))
    ax.plot([2,4,1,5], label="Label")
    ax.legend()
    # fig.tight_layout()
    fig.savefig(savepath)
    fig.show()

figsize=(2,2)
for i in range(1,4):
    f = savefolder / f'dpi_{i*72}.png'
    plot(figsize, i*72, f)

dpi=72
for i in [2,4,6]:
    f = savefolder / f'figsize_{i}_{i}.png'
    plot((i,i), dpi, f)

Reduce clutter

rng = np.random.default_rng()

x = np.linspace(0, 2, 100)
ys = []
for i in range(5):
    y = rng.normal(0,1,100)
    ys.append(y)
plt.style.use('default')

fig, ax = plt.subplots(dpi=150)
for i, y in enumerate(ys):
    ax.plot(x,y,label=f'Series {i+1}')
ax.legend()
f = savefolder / 'toomuchclutter.png'
fig.savefig(f, dpi=300)
plt.show()

# better
fig, axs = plt.subplots(nrows=len(ys), dpi=150, sharex=True)
for i in range(len(ys)):
    ax = axs[i]
    for j,y in enumerate(ys):
        if i==j:
            ax.plot(x,y, color='C0', label=f'Series {j}')
        else:
            ax.plot(x,y, color='C0', alpha=0.2)
    
    ax.set_ylabel(f'Series {i}', rotation=0, labelpad=20)
    ax.set_yticks([])
    ax.grid(True, axis='x', color='#9E9E9E')
    ax.spines['top'].set_visible(False)
    # ax.spines['right'].set_visible(False)
    ax.spines['bottom'].set_visible(False)


fig.tight_layout()
f = savefolder / 'reduce_clutter.png'
fig.savefig(f, dpi=300)
plt.show()
